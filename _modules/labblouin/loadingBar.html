import Tkinter


[docs]class loadingBar:
        
    # Create Progress Bar
    def __init__(self, title, width=300, height=25):
        self.obj = Tkinter.Tk()
        self.obj.resizable(False, False)
        self.obj.title(title)
        self.canvas = Tkinter.Canvas(self.obj, width=width, height=height)
        self.canvas.grid()
        self.width = width
        self.height = height
